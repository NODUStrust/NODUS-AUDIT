// src/components/LogList.jsx
import React from 'react';

function LogList({ logs, filter, loading, setSelectedLog }) {
  const filtered = logs.filter(
    (log) => !filter || log.action.toLowerCase().includes(filter.toLowerCase())
  );

  if (loading) return <div>Loading logs...</div>;
  if (filtered.length === 0) return <div>No logs found.</div>;

  return (
    <ul className="space-y-2 text-sm">
      {filtered.map((log, i) => (
        <li
          key={i}
          className="border-b py-2 cursor-pointer"
          onClick={() => setSelectedLog(log)}
        >
          <div>
            <strong>{log.action}</strong>{' '}
            <span className="text-gray-500 text-xs">
              [{new Date(log.timestamp).toLocaleString()}]
            </span>
          </div>
          <div className="text-gray-700">
            {log.result.length > 80 ? log.result.slice(0, 80) + '...' : log.result}
          </div>
          {log.status && (
            <div className="text-xs">
              Status:{' '}
              <span className={log.status === 'success' ? 'text-green-600' : 'text-red-500'}>
                {log.status === 'success' ? '✅ Success' : '❌ Failed'}
              </span>
            </div>
          )}
        </li>
      ))}
    </ul>
  );
}

// src/components/LogModal.jsx
import React from 'react';

function LogModal({ log, close }) {
  const downloadJSON = () => {
    const blob = new Blob([JSON.stringify(log, null, 2)], {
      type: 'application/json',
    });
    const url = URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = url;
    link.download = `log-${log.id || 'entry'}.json`;
    link.click();
    URL.revokeObjectURL(url);
  };

  return (
    <div className="fixed inset-0 bg-black bg-opacity-40 flex items-center justify-center z-50">
      <div className="bg-white p-4 rounded shadow max-w-2xl w-full space-y-4">
        <div className="flex justify-between items-center">
          <h4 className="font-semibold">Log Detail</h4>
          <div className="space-x-2">
            <button onClick={downloadJSON} className="text-sm text-blue-600 hover:underline">
              Download JSON
            </button>
            <button onClick={close} className="text-sm text-red-500 hover:underline">
              Close
            </button>
          </div>
        </div>
        <div className="text-sm whitespace-pre-wrap overflow-y-auto max-h-96">
          {log.result}
        </div>
      </div>
    </div>
  );
}

export default LogModal;

// src/components/Dashboard.jsx
import React, { useEffect, useState } from 'react';
import { signIn, onAuthStateChange } from '../auth';
import { startAction } from '../api';
import LogList from './LogList';
import LogModal from './LogModal';
import TriggerControls from './TriggerControls';

function Dashboard() {
  const [user, setUser] = useState(null);
  const [logs, setLogs] = useState([]);
  const [isRunning, setIsRunning] = useState(null);
  const [loadingLogs, setLoadingLogs] = useState(false);
  const [filter, setFilter] = useState(() =>
    new URLSearchParams(window.location.search).get('filter') || ''
  );
  const [selectedLog, setSelectedLog] = useState(null);

  useEffect(() => {
    const unsub = onAuthStateChange(setUser);
    return () => unsub();
  }, []);

  useEffect(() => {
    if (user) fetchLogs();
  }, [user]);

  useEffect(() => {
    const params = new URLSearchParams(window.location.search);
    filter ? params.set('filter', filter) : params.delete('filter');
    window.history.replaceState({}, '', `${window.location.pathname}?${params.toString()}`);
  }, [filter]);

  const fetchLogs = async () => {
    setLoadingLogs(true);
    try {
      // Replace this with real API call
      await new Promise((res) => setTimeout(res, 1000));
      setLogs([
        {
          id: 1,
          action: 'deploy',
          result: 'Deployed successfully',
          status: 'success',
          timestamp: Date.now(),
        },
        {
          id: 2,
          action: 'test',
          result: 'Test failed due to timeout error',
          status: 'failed',
          timestamp: Date.now() - 100000,
        },
      ]);
    } catch (err) {
      console.error('Failed to fetch logs', err);
    } finally {
      setLoadingLogs(false);
    }
  };

  const trigger = async (type) => {
    setIsRunning(type);
    try {
      const r = await startAction(type);
      alert(`Result: ${r.status || 'done'}`);
      fetchLogs();
    } catch (err) {
      alert(`Error: ${err.message}`);
    } finally {
      setIsRunning(null);
    }
  };

  if (!user) {
    return (
      <div className="p-4">
        <button onClick={signIn}>Sign in with Google</button>
      </div>
    );
  }

  return (
    <div className="p-4 space-y-4">
      <TriggerControls isRunning={isRunning} trigger={trigger} />

      <div className="flex items-center space-x-2">
        <label htmlFor="filter" className="text-sm">Filter by action:</label>
        <input
          id="filter"
          type="text"
          value={filter}
          onChange={(e) => setFilter(e.target.value)}
          className="border px-2 py-1 rounded"
        />
      </div>

      <h3 className="font-bold text-lg">Logs</h3>
      <LogList logs={logs} filter={filter} loading={loadingLogs} setSelectedLog={setSelectedLog} />

      {selectedLog && <LogModal log={selectedLog} close={() => setSelectedLog(null)} />}
    </div>
  );
}

export default Dashboard;
